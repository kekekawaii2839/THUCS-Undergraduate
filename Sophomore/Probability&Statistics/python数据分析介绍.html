Monty Hall Problem (三门问题)：三扇门后有两只山羊和一辆汽车，参赛者随机选择一扇门，选定后主持人开启余下两扇门的其中一扇，露出山羊，参赛者决定是否将自己的选择换到另一扇未开启的门

In [ ]:
import numpy as np
def monty_hall(n, change=True):
    doors = ["Car", "Goat", "Goat"]
    choices = []
    for i in range(n):
        choices.append(np.random.choice(doors, size=1)[0])
        if change:
            if choices[i] == "Car":
                choices[i] = "Goat"
            else:
                choices[i] = "Car"
    result = choices.count("Car")
    return result/n

np.random.seed(2022)

def demere_np(n):
    single = np.random.randint(1, 7, (n, 4))
    dobule1 = np.random.randint(1, 7, (n, 24))
    double2 = np.random.randint(1, 7, (n, 24))
    total_s = np.any(single == 6, axis=1).sum()
    total_d = np.any(dobule1 + double2 == 12, axis=1).sum()
    return total_s/n, total_d/n
In [ ]:
def demere(n):
    total_s = 0
    total_d = 0
    i = 0
    while i < n:
        for j in range(4):
            if np.random.randint(1,7) == 6:
                total_s += 1
                break
        for k in range(24):
            if np.random.randint(1,7) == 6 and np.random.randint(1,7) == 6:
                total_d += 1
                break
        i += 1
    return total_s/n, total_d/n
